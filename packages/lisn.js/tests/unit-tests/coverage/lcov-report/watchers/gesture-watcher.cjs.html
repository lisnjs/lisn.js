  const wrapper = (target, device, event, preventDefault) => {

      deltaThreshold,
      reverseScroll

      return true;
    }

      // afterwards and prevent default action.
      preventNextClick = true;
      setTimeout(() => {

      }, 10);
    }

    return false;
  };

    _callback: callback,
    _wrapper: wrapper
  };
